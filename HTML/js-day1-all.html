    function namesToHTMLList(namesArray) {
        const items = namesArray.map(name => `
  • ${name}
    • `).join(' ');
        return `
       ${items} 
    `;
        }
        
        const names = ["Lars", "Peter", "Jan", "Ian"];
        console.log(namesToHTMLList(names));
